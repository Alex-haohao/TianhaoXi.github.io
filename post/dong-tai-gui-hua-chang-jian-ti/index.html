function coinChange(coins,amount){
let dp = new Array(amount+1).fill(Infinity)
dp[0]=0;
for(let coin of coins){
for(let i =1;i<=amount;i++){
if(i-coin>=0){
dp[i]=Math.min(dp[i],dp[i-coin]+1)
}
}
}

return dp[amount] === Infinity ? 'Impossible' : dp[amount];

}

function minPath(grid){
let m = grid.length;
let n = grid[0].length;

for(let i =m-1;i>=0;i--){
    for(let j=n-1;j>=0;j--){
        if (i + 1 < m && j + 1 < n) {
            grid[i][j] += Math.min(grid[i + 1][j], grid[i][j + 1])
        } else if (i + 1 < m) {
            grid[i][j] += grid[i + 1][j]
        } else if (j + 1 < n) {
            grid[i][j] += grid[i][j + 1]
        }
    }
}
return grid[0][0];

}

function DyadicArray(N, c, w, p) {
//初始化表格
var dp = new Array(N + 1); //表格有n+1行
for (let n = 0; n < N + 1; n++) {
dp[n] = new Array(c + 1).fill(0); //每行有c+1列
}

for (let i = 1; i <= N; i++) {
for (let j = 0; j <= c; j++) {
if (j - w[i] >= 0) {
dp[i][j] = Math.max(dp[i - 1][j], dp[i - 1][j - w[i]] + p[i]);
} else {
dp[i][j] = dp[i - 1][j];
}
}
}
return(dp[N][c])
}

var longestCommonSubsequence = function(text1, text2) {
let n = text1.length;
let m = text2.length;
let dp = Array.from(new Array(n+1),() => new Array(m+1).fill(0));

for(let i =1;i<=n;i++){
    for(let j=1;j<=m;j++){
        if(text1[i-1]===text2[j-1]){
            dp[i][j] = 1+dp[i-1][j-1];
        }
        else{
            dp[i][j] = Math.max(dp[i-1][j],dp[i][j-1])
        }
    }
}
return dp[n][m]

};

function kanpSack(capacity, weights, values, n) {
var a,b,dp = []

//首先，初始化用于寻找解决方案的矩阵KS[n+1][capacity+1]
for (var i = 0; i <= n; i++) {
dp[i] = [];
}
for (var i = 0; i <= n; i++) {
for (var j = 0; j <= capacity; j++) {
if (i === 0 || j === 0) {
//让第一行和第一列为0，因为第一行和第一列表示容量为0或没有物品可装
dp[i][j] = 0;
} else if (weights[i - 1] > j) {
dp[i][j] = dp[i - 1][j]; //当前物品重量大于当前容量
} else {
//当前物品重量小于当前容量，有两个选择，装或者不装，取价值最大的
a = dp[i - 1][j - weights[i - 1]] + values[i - 1]; //装入这个物品后的总价值
b = dp[i - 1][j]; //不装
dp[i][j] = Math.max(a,b)
}
}

}
return dp[n][capacity];

}

var numTrees = function(n) {
let dp = new Array(n+1).fill(0);
dp[0] = 1
dp[1] = 1
for (let i = 2; i <= n; ++i) {
for (let j = 1; j <= i; ++j) {
dp[i] += dp[j-1] * dp[i-j];
}
}
return dp[n]
};

var lengthOfLongestSubstring = function(s) {

let max = 0;
let arr = [];
for(let i =0;i<s.length;i++){
    let index = arr.indexOf(s[i]);
    if(index !== -1){
        arr.splice(0,index+1)
    }
    arr.push(s[i]);
    max = Math.max(arr.length,max)
}
return max

};
